# ============================================================
# VALIDATION FUNCTION
# ============================================================
def validate_category(data_dict, category_name):
    """
    Perform complete statistical validation for a biomolecular category.
    
    Returns:
    --------
    dict with all validation metrics
    """
    print(f"\n{'='*80}")
    print(f"VALIDATION: {category_name.upper()} (N={len(data_dict)})")
    print(f"{'='*80}")
    
    # Calculate errors and predictions
    results = []
    for name, mass_real in data_dict.items():
        q = round(math.log(mass_real / L0) / math.log(A))
        mass_predicted = L0 * (A ** q)
        relative_error = abs(mass_real - mass_predicted) / mass_real * 100
        
        results.append({
            "Molecule": name,
            "Mass_Da": mass_real,
            "Predicted_Mass_Da": mass_predicted,
            "Relative_Error_%": relative_error,
            "Quantum_Number_q": q
        })
    
    df = pd.DataFrame(results)
    
    # 1. Basic statistics
    mean_error = df['Relative_Error_%'].mean()
    median_error = df['Relative_Error_%'].median()
    std_error = df['Relative_Error_%'].std()
    max_error = df['Relative_Error_%'].max()
    min_error = df['Relative_Error_%'].min()
    
    print(f"\n📊 Basic Statistics:")
    print(f"  Mean error: {mean_error:.2f}%")
    print(f"  Median error: {median_error:.2f}%")
    print(f"  Std deviation: {std_error:.2f}%")
    print(f"  Maximum error: {max_error:.2f}%")
    print(f"  Minimum error: {min_error:.2f}%")
    print(f"  Samples ≤5% error: {(df['Relative_Error_%'] <= 5).sum()}/{len(df)}")
    
    # 2. R² calculation
    SS_tot = np.sum((df['Mass_Da'] - np.mean(df['Mass_Da']))**2)
    SS_res = np.sum((df['Mass_Da'] - df['Predicted_Mass_Da'])**2)
    R2 = 1 - (SS_res / SS_tot)
    
    print(f"\n📈 Model Fit:")
    print(f"  R² (linear): {R2:.4f}")
    
    # 3. Residual uniformity test (Kolmogorov-Smirnov)
    residuals = []
    for mass in df['Mass_Da']:
        log_ratio = math.log(mass / L0)
        q = round(log_ratio / math.log(A))
        residual = (log_ratio / math.log(A)) - q
        residuals.append(residual)
    
    residuals = np.array(residuals)
    ks_statistic, ks_p_value = stats.kstest(residuals, 'uniform', args=(-0.5, 1.0))
    
    print(f"\n🔬 Residual Uniformity Test (Kolmogorov-Smirnov):")
    print(f"  KS Statistic: {ks_statistic:.4f}")
    print(f"  P-value: {ks_p_value:.4f}")
    if ks_p_value > 0.05:
        print(f"  ✅ Residuals follow uniform distribution (p > 0.05)")
    else:
        print(f"  ⚠️  Residuals deviate from uniform distribution")
    
    # 4. Scalability test (Pearson correlation)
    correlation, corr_p_value = stats.pearsonr(
        np.log10(df['Mass_Da']), 
        df['Relative_Error_%']
    )
    
    print(f"\n📏 Scalability Test (Error vs. Mass):")
    print(f"  Pearson correlation: {correlation:+.4f}")
    print(f"  P-value: {corr_p_value:.4f}")
    if abs(correlation) < 0.1 and corr_p_value > 0.05:
        print(f"  ✅ No correlation detected - model scales well")
    else:
        print(f"  ⚠️  Correlation detected - potential mass-dependent bias")
    
    # Compile results
    validation_results = {
        'Category': category_name,
        'N': len(df),
        'Mean_Error_%': mean_error,
        'Median_Error_%': median_error,
        'Std_Error_%': std_error,
        'Max_Error_%': max_error,
        'Min_Error_%': min_error,
        'Samples_le_5%': (df['Relative_Error_%'] <= 5).sum(),
        'R2': R2,
        'KS_Statistic': ks_statistic,
        'KS_P_Value': ks_p_value,
        'Pearson_r': correlation,
        'Pearson_p': corr_p_value
    }
    
    return validation_results
